        function acakSoal(soalArray) {
            const salinanSoal = [...soalArray];
            for (let i = salinanSoal.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [salinanSoal[i], salinanSoal[j]] = [salinanSoal[j], salinanSoal[i]];
            }
            return salinanSoal;
        }

        function mulaiQuiz(mode) {

            document.getElementById('quiz-area').style.display = 'block';

            const soalAcak = acakSoal(bankSoal[mode]);
            daftarSoal = soalAcak.slice(0, 5);
            tampilkanSoal();
        }
